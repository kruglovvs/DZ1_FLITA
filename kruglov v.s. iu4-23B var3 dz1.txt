#include <stdio.h>
#define MAS {0,1,11110,11,100,101,11111110,111,1000,100001,1001}
#define COUNT 11 /*возможно сделать через const sizeof(MAS)
        const int bin_mas[] = MAS;
        const int count = sizeof(mas [])/sizeof(mas[0]);*/

unsigned int to_dec(unsigned int bin){/*переводит десятичное число в двоичное*/
  unsigned int dec = 0;
  unsigned int degree = 1;
  while (bin != 0){
    dec += (bin%10)*degree;
    bin /= 10;
    degree *= 2;
  }
  return (dec);
}
int find_max_len(unsigned int bin_mas[]){/*находит максимальную длину максимального двоичного числа для красивого вывода*/
  unsigned int max = bin_mas[0];
  int max_len = 0;
  for (int i=1; i<COUNT; i++){
    if (max<bin_mas[i]){
      max = bin_mas[i];
    }
  }
  while (max != 0){
    max /= 10;
    max_len += 1;
  }
  return(max_len);
}
void print_mas(unsigned int bin_mas[], unsigned int dec_mas[]){/*выводит таблицу чисел из 2 массивов*/
  int row_len = find_max_len(bin_mas);
  for (int i=0; i<COUNT; i++){
    printf("%0*i -> %i\n",row_len,bin_mas[i],dec_mas[i]);
  }
}
int main(){
  unsigned int bin_mas[] = MAS;
  unsigned int dec_mas[COUNT];
  for (int i=0; i<COUNT; i++){
    dec_mas[i] = to_dec(bin_mas[i]);
  }
  print_mas(bin_mas,dec_mas);
}
